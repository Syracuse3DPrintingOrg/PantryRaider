# Host-side Bluetooth thermometer reader. bleak talks to bluez over D-Bus,
# so it works on a Raspberry Pi (or any Linux host) without native builds.
bleak>=0.21
httpx>=0.24
# The Cub status broadcast (advertiser.py) drives bluez advertising over
# D-Bus with dbus-fast, which bleak already pulls in on Linux; pinned here
# explicitly so the dependency is stated, not inherited by accident.
dbus-fast>=1.83; platform_system == "Linux"
# Plug-in STEMMA QT / Qwiic accessories (FoodAssistant-etsc) talk I2C through
# /dev/i2c-1. smbus2 is a tiny pure-Python wheel, and the import is lazy, so
# a host with no I2C bus (or no smbus2 installed) runs the reader unchanged.
smbus2>=0.4; platform_system == "Linux"
# A UART barcode scanner (FoodAssistant-x61t) wired to the Pi's GPIO serial
# port is read through pyserial. The import is lazy (inside SerialScanner.open),
# so a host with no scanner (or no pyserial installed) runs the reader
# unchanged and the tests need no serial hardware.
pyserial>=3.5; platform_system == "Linux"
